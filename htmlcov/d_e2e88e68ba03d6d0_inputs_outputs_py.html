1"""Github Actions helper functions. 

2 

3We want to support Python 3.7 that you still have on some self-hosted action runners. 

4So no fancy features like walrus operator, @cached_property, etc.

13INPUT_PREFIX = "INPUT_"

16class ActionInputs(EnvAttrDictVars): 

17 """GitHub Action input variables. 

18 

19 Usage: 

20 ```python 

21 class MyInputs(ActionInputs): 

22 my_input: str 

23 

24 action = ActionBase(inputs=MyInputs()) 

25 print(action.inputs.my_input) 

26 print(action.inputs["my-input"]) # the same as above 

27 ``` 

28 

29 Attribute names converted to `kebab-case`. 

30 So `action.inputs.my_input` is the same as `action.inputs["my-input"]`. 

31 

32 If you need to access `snake_case` named input `my_input` you should 

33 use dict-style only: `action.inputs["my_input"]`. 

34 But it's common to use `kebab-case` in GitHub Actions inputs names. 

35 

36 By github convention all input names are upper-cased in environment and prefixed with "INPUT_". 

37 So `actions.inputs.my_input` or `actions.inputs['my-input']` will be variable `INPUT_MY-INPUT` in environment. 

38 The ActionInputs does the conversion automatically. 

39 

40 Use lazy loading of the values. 

41 So the value is read from environment only when accessed and only once, and saved in the object internal dict. 

42 """ 

43 

44 # pylint: disable=abstract-method # we want RO implementation that raises NotImplementedError on write 

45 

46 def _external_name(self, name: str) -> str: 

47 """Convert variable name to the external form.""" 

48 return INPUT_PREFIX + name.upper()

51class ActionOutputs(FileAttrDictVars): 

52 """GitHub Actions output variables. 

53 

54 Usage: 

55 ```python 

56 class MyOutputs(ActionOutputs): 

57 my_output: str 

58 

59 action = ActionBase(outputs=MyOutputs()) 

60 action.outputs["my-output"] = "value" 

61 action.outputs.my_output = "value" # the same as above 

62 ``` 

63 

64 Attribute names converted to `kebab-case`. 

65 So `action.outputs.my_output` is the same as `action.outputs["my-output"]`. 

66 

67 If you need to access `snake_case` named output `my_output` you should 

68 use dict-style only: `action.outputs["my-output"]`. 

69 But it's common to use `kebab-case` in GitHub Actions outputs names. 

70 

71 Each outputs vars assignment change the Github outputs file (the path is defined as `action.env.github_output`). 

72 """ 

73 

74 def __init__(self) -> None: 

75 super().__init__(Path(os.environ["GITHUB_OUTPUT"]))
